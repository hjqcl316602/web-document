# 基础

## 查看所有的数据库

show databases;

## 进行制定的数据库

use [mysql]

## 查询数据下的所有的表

show tables;

## 查询表

select \* from [user]

## 查看默认的 MYSQL 用户

use mysql;
select \* from user

## 新增数据

INSERT into t_mode(name,code) values('欠款','mo') ; 存在自增长的 id，不写即可

## 更新数据

UPDATE table_name SET field1=new-value1, field2=new-value2 [WHERE Clause]

## 删除数据

DELETE from [t_mode] where id = 3
